<body>
    <h1>利用 WebWorker 计算质数数量</h1>
    <div>
        <label for="start">输入范围起始：</label>
        <input type="number" id="start" value="1" />
        <label for="end">输入范围结束：</label>
        <input type="number" id="end" value="1000000" />
        <button id="startButton">开始计算</button>
    </div>

    <div id="progress">等待开始计算...</div>
    <div id="result"></div>

    <hr />

    <h2>UI 动态效果展示（平移动画）</h2>
    <div id="container">
        <div id="circle1" class="circle"></div>
        <div id="circle2" class="circle"></div>
        <div id="circle3" class="circle"></div>
    </div>
    
    <button id="startAnimationButton">开始动画</button>

    <script>
      let isRunning = false;
      let stopRequested = false; // 标记停止请求
      let circlePositions = [0, 0, 0]; // 保存每个球的当前水平位置

      // 启动按钮的点击事件
      // TODO 优化计算质数的逻辑，放入worker.js中，随后进行onMessage、postMessage的交互。
      // tip: worker也可以传输一个自定义的对象，可以增加一些字段属性用于表示是过程还是结果、或者用于表示具体内容。
      // 参考学习：https://developer.mozilla.org/zh-CN/docs/Web/API/Web_Workers_API/Using_web_workers#web_worker_api
      document.getElementById("startButton").addEventListener("click", () => {
        const start = parseInt(document.getElementById("start").value, 10);
        const end = parseInt(document.getElementById("end").value, 10);
        if (
          isNaN(start) ||
          isNaN(end) ||
          start <= 0 ||
          end <= 0 ||
          end < start
        ) {
          alert("请输入有效的范围！");
          return;
        }

        // 质数计算函数
        function calculatePrimes(start, end) {
          let count = 0;
          for (let num = start; num <= end; num++) {
            if (isPrime(num)) {
              count++;
            }
          }
          return count;
        }

        // TODO 检查一个数是否为质数 (可优化，最终均需要放到worker.js中)
        function isPrime(num) {
          for (let i = 2; i <= num; i++) {
            if (num % i === 0) {
              return false;
            }
          }
          return true;
        }

        document.getElementById("progress").textContent = "正在计算，请稍候..."; 
        document.getElementById("result").textContent = ""; 

        const primeCount = calculatePrimes(start, end); // TODO 优化：过程显示当前计算的数字 '正在计算:' + number;
        document.getElementById("progress").textContent = "计算完成！";
        document.getElementById(
          "result"
        ).textContent = `范围内的质数数量是：${primeCount}`;
      });

      // TODO 平移动画 只需要修重新点击开始动画后的重置bug，但是不要改移动的逻辑。校验将会测试小球的left属性判定位置移动情况。
      function startAnimation() {
        const circles = [
          document.getElementById("circle1"),
          document.getElementById("circle2"),
          document.getElementById("circle3"),
        ];
        circles.forEach((circle, index) => {
          let position = 0;
          circlePositions[index] = 0; // 重置位置

          const interval = setInterval(() => {
            // 每次动画开始时重新设置位置
            if (position > window.innerWidth) {
              circle.style.left = `${window.innerWidth}px`; // 设置静止在右侧
              clearInterval(interval); // 动画结束时停止...
            } else {
              position += 1;
              circle.style.left = `${position}px`;
              circlePositions[index] = position; // 更新球的位置
            }
          }, 5); 
        });
      }

      // 控制动画的重置
      document
        .getElementById("startAnimationButton")
        .addEventListener("click", () => {
          const circles = [
            document.getElementById("circle1"),
            document.getElementById("circle2"),
            document.getElementById("circle3"),
          ];
          circles.forEach((circle) => {
            circle.style.left = "0px"; 
          });
          startAnimation(); // 重新开始动画
        });
    </script>
